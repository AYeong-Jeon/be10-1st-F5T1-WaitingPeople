SELECT
		rv.review_content
	  , rv.review_rating
	  , rv.mod_date
	  , imf.picture_name
	  , imf.picture_original_name

	FROM tb_review rv

 WHERE rv.review_seq = '3';

-- 내가 쓴 리뷰 삭제 by ayeong
START TRANSACTION;

ALTER TABLE tb_image_file DROP FOREIGN KEY tb_image_file_FK;

ALTER TABLE tb_image_file
ADD CONSTRAINT tb_image_file_FK FOREIGN KEY (review_seq)
REFERENCES tb_review (review_seq)
ON DELETE CASCADE;

ALTER TABLE tb_review_comment DROP FOREIGN KEY tb_review_comment_FK;

ALTER TABLE tb_review_comment
ADD CONSTRAINT tb_review_comment_FK FOREIGN KEY (review_seq)
REFERENCES tb_review (review_seq)
ON DELETE CASCADE;

DELETE FROM tb_review
 WHERE review_seq = '3';

COMMIT;

SELECT
		rv.review_content
	  , rv.review_rating
	  , rv.mod_date
	  , imf.picture_name
	  , imf.picture_original_name
	  , imf.picture_url
	  , rvc.comment_content
	FROM  tb_image_file imf
	JOIN tb_review rv ON imf.review_seq = rv.review_seq
	JOIN tb_review_comment rvc ON rv.review_seq = rvc.review_seq
 WHERE rv.review_seq = '3';
